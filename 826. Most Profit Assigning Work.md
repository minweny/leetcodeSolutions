class Solution:
    def maxProfitAssignment(self, difficulty: List[int], profit: List[int], worker: List[int]) -> int:
        l = len(difficulty)
        arr = sorted([[difficulty[i], profit[i]] for i in range(l)])
        maxProfit = 0
        for i in range(l):
            maxProfit = max(maxProfit, arr[i][1])
            arr[i][1] = maxProfit
        # print(arr)
        res = 0
        for w in worker:
            left, right = 0, l
            while left < right:
                m = (left + right) // 2
                if arr[m][0] <= w:
                    left = m + 1
                else:
                    right = m
            # print(f"w: {w}, left: {left} ---------------")
            if left > 0:
                res += arr[left - 1][1]
                # print(f"profit: {arr[left - 1][1]}")
        return res
