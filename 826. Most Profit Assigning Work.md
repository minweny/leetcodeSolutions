dp - O(n)
```
class Solution:
    def maxProfitAssignment(self, difficulty: List[int], profit: List[int], worker: List[int]) -> int:
        maxD = max(difficulty)
        dp = [0] * (maxD + 1)
        for i in range(len(difficulty)):
            dp[difficulty[i]] = max(dp[difficulty[i]], profit[i])
        maxPro = 0
        for i in range(maxD + 1):
            maxPro = max(maxPro, dp[i])
            dp[i] = maxPro
        res = 0
        for w in worker:
            if w > maxD:
                res += dp[-1]
            else:
                res += dp[w]
        return res
```

binary search - O(n*log(n))
```
class Solution:
    def maxProfitAssignment(self, difficulty: List[int], profit: List[int], worker: List[int]) -> int:
        l = len(difficulty)
        arr = sorted([[difficulty[i], profit[i]] for i in range(l)])
        maxProfit = 0
        for i in range(l):
            maxProfit = max(maxProfit, arr[i][1])
            arr[i][1] = maxProfit
        # print(arr)
        res = 0
        for w in worker:
            left, right = 0, l
            while left < right:
                m = (left + right) // 2
                if arr[m][0] <= w:
                    left = m + 1
                else:
                    right = m
            # print(f"w: {w}, left: {left} ---------------")
            if left > 0:
                res += arr[left - 1][1]
                # print(f"profit: {arr[left - 1][1]}")
        return res
```

```
class Solution:
    def maxProfitAssignment(self, difficulty: List[int], profit: List[int], worker: List[int]) -> int:
        l = len(difficulty)
        arr0 = sorted([[difficulty[i], profit[i]] for i in range(l)])
        arr = [arr0[0]]
        for i in range(1, l):
            if arr0[i][1] > arr[-1][1]:
                arr.append(arr0[i])
        # print(arr)
        res = 0
        for w in worker:
            left, right = 0, len(arr)
            while left < right:
                m = (left + right) // 2
                if arr[m][0] <= w:
                    left = m + 1
                else:
                    right = m
            # print(f"w: {w}, left: {left} ---------------")
            if left > 0:
                res += arr[left - 1][1]
                # print(f"profit: {arr[left - 1][1]}")
        return res
```
