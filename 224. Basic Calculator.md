class Solution:
    def calculate(self, s: str) -> int:
        stack = []
        pre = cur = 0
        op = 1
        for c in s:
            if c == " ":
                continue
            elif c == "(":
                stack.append(pre)
                stack.append(op)
                pre = 0
                op = 1
            elif c.isdigit():
                cur = 10 * cur + int(c)
            elif c in ("+", "-", ")"):
                pre = pre + op * cur
                cur = 0
                if c == "+":
                    op = 1
                elif c == "-":
                    op = -1
                else:
                    cur = pre
                    op = stack.pop()
                    pre = stack.pop()
                    pre = pre + op * cur
                    cur = 0
            # print(f"c: {c}, pre: {pre}, op: {op}, cur: {cur}, stack:{stack}")
        if cur != 0:
            pre = pre + op * cur
            cur = 0
            # print(f"c: {c}, pre: {pre}, op: {op}, cur: {cur}, stack:{stack}")
        return pre
