class Solution:
    def solveSudoku(self, board: List[List[str]]) -> None:
        """
        Do not return anything, modify board in-place instead.
        """
        rowSets = [set() for i in range(9)]
        colSets = [set() for i in range(9)]
        boxSets = [set() for i in range(9)]
        emptyCells = []
        for i in range(9):
            for j in range(9):
                n = board[i][j]
                if n == ".":
                    emptyCells.append((i, j))
                else:
                    rowSets[i].add(n)
                    colSets[j].add(n)
                    ii = i // 3
                    jj = j // 3
                    boxSets[ii * 3 + jj].add(n)
        # print(emptyCells, len(emptyCells))
        def backtrack(idx):
            if idx == len(emptyCells):
                return True
            i, j = emptyCells[idx]
            ii = i // 3
            jj = j // 3
            for n in range(1, 10):
                nn = str(n)
                if nn in rowSets[i] or nn in colSets[j] or nn in boxSets[ii * 3 + jj]:
                    continue
                board[i][j] = nn
                rowSets[i].add(nn)
                colSets[j].add(nn)
                boxSets[ii * 3 + jj].add(nn)

                if backtrack(idx + 1):
                    return True

                board[i][j] = "."
                rowSets[i].remove(nn)
                colSets[j].remove(nn)
                boxSets[ii * 3 + jj].remove(nn)
            return False
        backtrack(0)
