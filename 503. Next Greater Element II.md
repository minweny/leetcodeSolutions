class Solution:
    def nextGreaterElements(self, nums: List[int]) -> List[int]:
        l = len(nums)
        stack = []
        for i in range(l - 1, -1, -1):
            num = nums[i]
            while stack and num >= stack[-1]:
                stack.pop()
            stack.append(num)
        res = []
        for i in range(l - 1, -1, -1):
            num = nums[i]
            while stack and num >= stack[-1]:
                stack.pop()
            if stack:
                res.append(stack[-1])
            else:
                res.append(-1)
            stack.append(num)
        return res[::-1]
