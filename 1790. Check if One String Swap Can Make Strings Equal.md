```
class Solution:
    def areAlmostEqual(self, s1: str, s2: str) -> bool:
        l = len(s1)
        dif = []
        for i in range(l):
            if s1[i] != s2[i]:
                dif.append(i)
        if len(dif) > 2 or len(dif) == 1: return False
        if len(dif) == 0: return True
        i, j = dif[0], dif[1]
        return s1[i] == s2[j] and s1[j] == s2[i]
```
