1. bucket sort - O(n)
```
class Solution:
    def findKthLargest(self, nums: List[int], k: int) -> int:
        bucket = [0] * 20001
        # -10000, 0, 10000 -> 0, 10000, 20000
        for n in nums:
            bucket[n + 10000] += 1
        for i in range(20000, -1, -1):
            if k <= bucket[i]:
                # print(f"i = {i}")
                return i - 10000
            k -= bucket[i]
        return -1
```
