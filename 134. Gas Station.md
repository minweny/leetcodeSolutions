class Solution:
    def canCompleteCircuit(self, gas: List[int], cost: List[int]) -> int:
        if sum(gas) < sum(cost):
            return -1
        l = len(gas)
        start = 0
        rem = 0
        for i in range(l):
            rem += gas[i] - cost[i]
            if rem < 0:
                start = i + 1
                rem = 0
        return start if start < l else -1
