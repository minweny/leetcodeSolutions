O(n)
```
class Solution:
    def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]:
        ansMap = {}
        len2 = len(nums2)
        stack = []
        for i in range(len2 - 1, -1, -1):
            num = nums2[i]
            while stack and num >= stack[-1]:
                stack.pop()
            if stack:
                ansMap[num] = stack[-1]
            else:
                ansMap[num] = -1
            stack.append(num)
        res = []
        for num in nums1:
            res.append(ansMap[num])
        return res
```
