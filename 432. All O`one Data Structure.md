class Node:
    def __init__(self, fre=0):
        self.fre = fre
        self.s = set()
        self.pre = None
        self.nex = None

class AllOne:
    def __init__(self):
        self.head = Node()
        self.tail = Node()
        self.head.nex = self.tail
        self.tail.pre = self.head

        self.strToNode = {}

    def removeNode(self, node):
        pre = node.pre
        nex = node.nex
        pre.nex = nex
        nex.pre = pre

        return pre

    def insertNode(self, pre, fre):
        node = Node(fre)
        nex = pre.nex

        pre.nex = node
        node.pre = pre

        node.nex = nex
        nex.pre = node

        return node

    def inc(self, key: str) -> None:
        # print(f"inc: {key}")
        if key in self.strToNode:
            oldNode = self.strToNode[key]
            oldNode.s.remove(key)
            oldFre = oldNode.fre
            
            if len(oldNode.s) == 0:
                pre = self.removeNode(oldNode)
            else:
                pre = oldNode
        else:
            oldFre = 0
            pre = self.head
        
        newFre = oldFre + 1
        if newFre == pre.nex.fre:
            node = pre.nex
        else:
            node = self.insertNode(pre, newFre)
        node.s.add(key)
        self.strToNode[key] = node
        # self.print()


    def dec(self, key: str) -> None:
        # print(f"dec: {key}")
        oldNode = self.strToNode[key]
        oldNode.s.remove(key)
        oldFre = oldNode.fre
        if len(oldNode.s) == 0:
            pre = self.removeNode(oldNode)
        else:
            # > 1
            pre = oldNode.pre

        newFre = oldFre - 1
        if newFre > 0:
            if newFre == pre.fre:
                node = pre
            else:
                node = self.insertNode(pre, newFre)
            node.s.add(key)
            self.strToNode[key] = node
        else:
            del self.strToNode[key]

    def getMaxKey(self) -> str:
        # print(f"getMaxKey")
        last = self.tail.pre
        if len(last.s) > 0:
            return list(last.s)[0]
        else:
            return ""

    def getMinKey(self) -> str:
        # print(f"getMinKey")
        first = self.head.nex
        if len(first.s) > 0:
            return list(first.s)[0]
        else:
            return ""

    def print(self):
        node = self.head
        while node:
            if node != self.head and node != self.tail:
                print(node.fre, node.s)
            node = node.nex
