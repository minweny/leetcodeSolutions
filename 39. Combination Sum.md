class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        candidates = sorted(list(set(candidates)))
        l = len(candidates)
        res = []
        def backtrack(i, nums, total):
            for j in range(i, l):
                if candidates[j] + total == target:
                    nums.append(candidates[j])
                    res.append(list(nums))
                    nums.pop()
                elif candidates[j] + total < target:
                    nums.append(candidates[j])
                    backtrack(j, nums, candidates[j] + total)
                    nums.pop()
        backtrack(0, [], 0)
        return res
